## Example
```c
#include <stdio.h>
#include <conio.h>


int main() {
	int min = arr[0];
	int max = arr[0];

	printf("\n");
	for (int i = 0; i < 5; i++) {
		if(arr[i] < min)
			min = arr[i];
		if(arr[i] > max)
			max = arr[i];
	}
	printf("Minimum: %d\n", min);
	printf("Maximum: %d\n", max);

	return 0;
}
```
